local function getAllTargets()
    local targets = {}
    local useBots = Toggles.BotSupport and Toggles.BotSupport.Value
    
    -- Insert online players
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            table.insert(targets, p.Character)
        end
    end
    
    -- Insert non-player NPC bots if toggle enabled
    if useBots then
        for _, child in pairs(workspace:GetChildren()) do
            if child:IsA("Model") and child ~= LP.Character and not Players:GetPlayerFromCharacter(child) then
                local hum = child:FindFirstChildOfClass("Humanoid")
                local root = child:FindFirstChild("HumanoidRootPart")
                if hum and root and hum.Health > 0 then
                    table.insert(targets, child)
                end
            end
        end
    end
    return targets
end

local function getClosestInFOV()
    local closest = nil
    local closestPart = nil
    local shortestDist = math.huge
    local center = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    local radius = Options['Fov Circle'].Value

    local targets = getAllTargets()
    for _, char in pairs(targets) do
        local hum = char:FindFirstChildOfClass("Humanoid")
        local root = char:FindFirstChild("HumanoidRootPart")
        if hum and root and hum.Health > 0 then
            local sp, on = Camera:WorldToViewportPoint(root.Position)
            if on and sp.Z > 0 then
                local dist = (Vector2.new(sp.X, sp.Y) - center).Magnitude
                if dist <= radius and dist < shortestDist then
                    closest = char
                    closestPart = root
                    shortestDist = dist
                end
            end
        end
    end
    return closest, closestPart
end

local function isCharacterAlive(char)
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum and hum.Health > 0 then
            return true
        end
    end
    return false
end

local function checkWall(part)
    if not Toggles.ToggleWallCheck or not Toggles.ToggleWallCheck.Value then return true end
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = {LP.Character, Camera, part.Parent}
    
    local dir = (part.Position - Camera.CFrame.Position).Unit * (part.Position - Camera.CFrame.Position).Magnitude
    local hit = workspace:Raycast(Camera.CFrame.Position, dir, params)
    
    return hit == nil 
end

local function getNearestPart(char)
    if not Toggles.ToggleNearestPoint or not Toggles.ToggleNearestPoint.Value then
        return char:FindFirstChild("HumanoidRootPart")
    end
    local closestPart = nil
    local shortest = math.huge
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    
    for _, v in pairs(char:GetChildren()) do
        if v:IsA("BasePart") then
            local pos, on = Camera:WorldToViewportPoint(v.Position)
            if on then
                local dist = (Vector2.new(pos.X, pos.Y) - screenCenter).Magnitude
                if dist < shortest then
                    closestPart = v
                    shortest = dist
                end
            end
        end
    end
    return closestPart or char:FindFirstChild("HumanoidRootPart")
end

local function getLockTarget()
    local closest = nil
    local shortest = math.huge
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    local targets = getAllTargets()
    for _, char in pairs(targets) do
        local root = char:FindFirstChild("HumanoidRootPart")
        if root then
            local pos, on = Camera:WorldToViewportPoint(root.Position)
            if on and pos.Z > 0 then
                local dist = (Vector2.new(pos.X, pos.Y) - screenCenter).Magnitude
                if dist < shortest then
                    if checkWall(root) then
                        closest = char
                        shortest = dist
                    end
                end
            end
        end
    end
    return closest
end

-- Render Stepped Loop
RS.RenderStepped:Connect(function()
    -- Aimlock logic
    if Toggles.ToggleEnabled and Toggles.ToggleEnabled.Value then
        if not CurrentTarget or not isCharacterAlive(CurrentTarget) then
            CurrentTarget = getLockTarget()
        end

        if CurrentTarget then
            local targetPart = getNearestPart(CurrentTarget)

            if Toggles.ToggleViewTarget and Toggles.ToggleViewTarget.Value then
                Camera.CameraSubject = CurrentTarget:FindFirstChildOfClass("Humanoid")
            else
                Camera.CameraSubject = LP.Character:FindFirstChildOfClass("Humanoid")
            end

            if Toggles.ToggleLookAtTarget and Toggles.ToggleLookAtTarget.Value and targetPart then
                local rootPart = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
                if rootPart then
                    local targetPos = Vector3.new(targetPart.Position.X, rootPart.Position.Y, targetPart.Position.Z)
                    rootPart.CFrame = CFrame.new(rootPart.Position, targetPos)
                end
            end

            if targetPart and checkWall(targetPart) then
                local pred = tonumber(Options.PredictionInput.Value) or 0
                local targetPos = targetPart.Position + (targetPart.AssemblyLinearVelocity * pred)
                local smoothnessVal = Options.SmoothnessSlider and Options.SmoothnessSlider.Value or 1
                local smooth = math.clamp(1 / smoothnessVal, 0.01, 1)
                
                local currCFrame = Camera.CFrame
                local targetCFrame = CFrame.new(currCFrame.Position, targetPos)
                Camera.CFrame = currCFrame:Lerp(targetCFrame, smooth)
            end
        else
            CurrentTarget = nil
            if LP.Character and LP.Character:FindFirstChildOfClass("Humanoid") then
                Camera.CameraSubject = LP.Character:FindFirstChildOfClass("Humanoid")
            end
        end
    else
        CurrentTarget = nil
        if LP.Character and LP.Character:FindFirstChildOfClass("Humanoid") then
            Camera.CameraSubject = LP.Character:FindFirstChildOfClass("Humanoid")
        end
    end

    -- Silent Aim logic
    if Toggles.ToggleSilentAim and Toggles.ToggleSilentAim.Value then
        fovCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
        fovCircle.Radius = Options['Fov Circle'] and Options['Fov Circle'].Value or 150
        fovCircle.Visible = true
        local sTarget, sPart = getClosestInFOV()
        silentTarget = sTarget
        silentTargetPart = sPart or (sTarget and sTarget:FindFirstChild("HumanoidRootPart"))
    else
        fovCircle.Visible = false
        silentTarget = nil
        silentTargetPart = nil
    end
end)
